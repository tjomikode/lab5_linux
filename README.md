#OPTIONAL ARGUMENT -> (-s or --size)

	#size with value -> (-s100 or -size100 or --size=100)
#REQUIRED ARGUMENT -> (-f or --file)
	file = example:
		#-f example
		#-fexample
		#--file example
		#--file=example
